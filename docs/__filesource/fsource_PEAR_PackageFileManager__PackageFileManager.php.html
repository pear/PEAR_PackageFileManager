28    /**

29     * PEAR :: PackageGenerate updates the <filelist></filelist> section
30     * of a PEAR package.xml file to reflect the current files in
31     * preparation for a release.
32     *

33     * The PEAR_PackageGenerate class uses a plugin system to generate the
34     * list of files in a package.  This allows both standard recursive
35     * directory parsing (plugin type file) and more intelligent options
36     * such as the CVS browser {@link PEAR_PackageFileManager_Cvs}, which
37     * grabs all files in a local CVS checkout to create the list, ignoring
38     * any other local files.
39     *

40     * Other options include specifying roles for file extensions (all .php
41     * files are role="php", for example), roles for directories (all directories
42     * named "tests" are given role="tests" by default), and exceptions.
43     * Exceptions are specific pathnames with * and ? wildcards that match
44     * a default role, but should have another.  For example, perhaps
45     * a debug.tpl template would normally be data, but should be included
46     * in the docs role.  Along these lines, to exclude files entirely,
47     * use the ignore option.
48     *

49     * Required options for a release include version, baseinstalldir, state,
50     * and packagedirectory (the full path to the local location of the
51     * package to create a package.xml file for)
52     *

53     * Example usage:
54     * <code>
55     * <?php
56     * require_once('PEAR/PackageFileManager.php');
57     * $packagexml = new PEAR_PackageFileManager(
58     * array('baseinstalldir' => 'PhpDocumentor',
59     *  'version' => '1.2.1',
60     *  'packagedirectory' => 'C:/Web Pages/chiara/phpdoc2/',
61     *  'state' => 'stable',
62     *  'filelistgenerator' => 'cvs', // generate from cvs, use file for directory
63     *  'notes' => 'We\'ve implemented many new and exciting features',
64     *  'ignore' => array('TODO', 'tests/'), // ignore TODO, all files in tests/
65     *  'installexceptions' => array('phpdoc' => '/*'), // baseinstalldir ="/" for phpdoc
66     *  'dir_roles' => array('tutorials' => 'doc'),
67     *  'exceptions' => array('README' => 'doc', // README would be data, now is doc
68     *                        'PHPLICENSE.txt' => 'doc'))); // same for the license
69     * $packagexml->addRole('pkg', 'doc'); // add a new role mapping
70     * $packagexml->writePackageFile();
71     * ?>
72     * </code>
73     * @package PEAR_PackageFileManager
74     */

126       function PEAR_PackageFileManager($options = array())

127       {

128           if (!isset($options['state'])) {

129               return false;

130           }

131           if (!isset($options['version'])) {

132               return false;

133           }

134           if (!isset($options['packagedirectory'])) {

135               return false;

136           } else {

137               $options['packagedirectory'] = str_replace(DIRECTORY_SEPARATOR,

138                                                        '/',

139                                                        realpath($options['packagedirectory']));

140               if ($options['packagedirectory']{strlen($options['packagedirectory']) - 1} != '/') {

141                   $options['packagedirectory'] .= '/';

142               }

143           }

144           if (!isset($options['baseinstalldir'])) {

145               return false;

146           }

147           $this->_options = array_merge($this->_options, $options);

148           

149           $path = ($this->_options['pathtopackagefile'] ?

150                       $this->_options['pathtopackagefile'] : $this->_options['packagedirectory']);

151           $this->_options['filelistgenerator'] = ucfirst(strtolower($this->_options['filelistgenerator']));

152           if (!($res = $this->_getExistingPackageXML($path, $this->_options['packagefile']))) {

153               return $res;

154           }

155           // attempt to load the interface from the standard PEAR location

156           @include_once('PEAR/PackageFileManager/' . $this->_options['filelistgenerator'] . '.php');

157           if (!class_exists('PEAR_PackageFileManager_' . $this->_options['filelistgenerator'])) {

158               if (isset($this->_options['usergeneratordir'])) {

159                   // attempt to load from a user-specified directory

160                   $this->_options['usergeneratordir'] = str_replace(DIRECTORY_SEPARATOR,

161                                                                     '/',

162                                                                     realpath($this->_options['usergeneratordir']));

163                   if ($this->_options['usergeneratordir']{strlen($this->_options['usergeneratordir']) - 1} != '/') {

164                       $this->_options['usergeneratordir'] .= '/';

165                   }

166                   @include_once($this->_options['usergeneratordir'] . $this->_options['filelistgenerator'] . '.php');

167                   if (!class_exists('PEAR_PackageFileManager_' . $this->_options['filelistgenerator'])) {

168                       return false;

169                   }

170               } else {

171                   return false;

172               }

173           }

174       }

186       /**

187        * Add a dependency on another package, or an extension/php
188        *

189        * This will overwrite an existing dependency if it is found.  In
190        * other words, if a dependency on PHP 4.1.0 exists, and
191        * addDependency('php', '4.3.0', 'ge', 'php') is called, the existing
192        * dependency on PHP 4.1.0 will be overwritten with the new one on PHP 4.3.0
193        * @param string Dependency element name
194        * @param string Dependency version
195        * @param string A specific operator for the version, this can be one of:
196        *    'has', 'not', 'lt', 'le', 'eq', 'ne', 'ge', or 'gt'
197        * @param string Dependency type.  This can be one of:
198        *    'pkg', 'ext', 'php', 'prog', 'os', 'sapi', or 'zend'
199        */

200       function addDependency($name, $version = false, $operator = 'ge', $type = 'pkg')

201       {

202           $found = false;

203           foreach($this->_options['deps'] as $index => $dep) {

204               if ($dep['name'] == $name && $deps['type'] == $type) {

205                   $found = $index;

206                   break;

207               }

208           }

209           $dep =

210               array(

211                   'name' => $name,

212                   'type' => $type);

213           if ($version) {

214               $dep['version'] = $version;

215               if ($operator) {

216                   $dep['rel'] = $operator;

217               }

218           }

219   

220           if ($found !== false) {

221               $this->_options['deps'][$found] = $dep; // overwrite existing dependency

222           } else {

223               $this->_options['deps'][] = $dep; // add new dependency

224           }

225       }

227       /**

228        * Writes the package.xml file out with the newly created <release></release> tag
229        * @param boolean null if no debugging, true if web interface, false if command-line
230        */

231       function writePackageFile($debuginterface = null)

232       {

233           extract($this->_options);

234           $date = date('Y-m-d');

235           $this->_packageXml['release_date'] = $date;

236           $this->_packageXml['release_version'] = $version;

237           $this->_packageXml['release_state'] = $state;

238           $this->_packageXml['release_notes'] = $notes;

239           $this->_packageXml['filelist'] = $this->_getFileList();

240           $this->_packageXml['release_deps'] = $this->_getDependencies();

241           $this->_updateChangeLog();

242           $common = new PEAR_Common;

243           $packagexml = $common->xmlFromInfo($this->_packageXml);

244           if (isset($debuginterface)) {

245               if ($debuginterface) {

246                   echo '<pre>' . htmlentities($packagexml) . '</pre>';

247               } else {

248                   echo $packagexml;

249               }

250               return true;

251           }

252           $outputdir = ($this->_options['outputdirectory'] ?

253                           $this->_options['outputdirectory'] : $this->_options['packagedirectory']);

254           if (is_writable($this->_options['packagedirectory'] . $this->_options['packagefile'])) {

255               if ($fp = @fopen($outputdir . $this->_options['packagefile'] . '.tmp', "w")) {

256                   $written = @fwrite($fp, $packagexml);

257                   @fclose($fp);

258                   if ($written === false) {

259                       return false;

260                   }

261                   if (!@copy($outputdir . $this->_options['packagefile'] . '.tmp',

262                           $outputdir . $this->_options['packagefile'])) {

263                       return false;

264                   } else {

265                       @unlink($outputdir . $this->_options['packagefile'] . '.tmp');

266                       return true;

267                   }

268               } else {

269                   return false;

270               }

271           } else {

272               return false;

273           }

274       }

299       /**

300        * Recursively generate the <filelist> section's <dir> and <file> tags
301        * @param array the sorted directory structure
302        * @param false|stringwhether the parent directory has a role this should
303        *  inherit
304        * @param integer indentation level
305        * @access private
306        */

307       function _getDirTag($struc, $role=false)

308       {

309           extract($this->_options);

310           $ret = array();

311           foreach($struc as $dir => $files) {

312               if ($dir === '/') {

313                   return $this->_getDirTag($struc[$dir], $role);

314               } else {

315                   if (!isset($files['file'])) {

316                       $myrole = '';

317                       if ($role) {

318                           $myrole = $role;

319                       } elseif (isset($dir_roles[$dir])) {

320                           $myrole = $dir_roles[$dir];

321                       }

322                       $ret = array_merge($ret, $this->_getDirTag($files, $myrole));

323                   } else {

324                       $myrole = '';

325                       if (!$role)

326                       {

327                           $myrole = false;

328                           if (isset($exceptions[$files['file']])) {

329                               $myrole = $exceptions[$files['file']];

330                           } elseif (isset($roles[$files['ext']])) {

331                               $myrole = $roles[$files['ext']];

332                           } else {

333                               $myrole = $roles['*'];

334                           }

335                       } else {

336                           $myrole = $role;

337                       }

338                       if (isset($installexceptions[$files['file']])) {

339                           $bi = $installexceptions[$files['file']];

340                       } else {

341                           $bi = $baseinstalldir;

342                       }

343                       $ret[$files['path']] = array('role' => $myrole, 'baseinstalldir' => $bi);

344                   }

345               }

346           }

347           return $ret;

348       }

364       /**

365        * Creates a changelog entry with the current release
366        * notes and dates, or overwrites a previous creation
367        * @access private
368        */

369       function _updateChangeLog()

370       {

371           $curlog = false;

372           foreach($this->_packageXml['changelog'] as $index => $changelog) {

373               if ($changelog['version'] == $this->_options['version']) {

374                   $curlog = $index;

375               }

376           }

377           $notes = ($this->_options['changelognotes'] ?

378                       $this->_options['changelognotes'] : $this->_options['notes']);

379           $changelog = array('version' => $this->_options['version'],

380                              'release_date' => date('Y-m-d'),

381                              'release_license' => $this->_options['license'],

382                              'release_state' => $this->_options['state'],

383                              'release_notes' => $notes,

384                              );

385           if ($curlog !== false) {

386               $this->_packageXml['changelog'][$curlog] = $changelog;

387           } else {

388               $this->_packageXml['changelog'][] = $changelog;

389           }

390       }

395       function _getExistingPackageXML($path, $packagefile = 'package.xml')

396       {

397           if (@is_dir($path)) {

398               $contents = @file_get_contents($path . $packagefile);

399               if (!$contents) {

400                   return false;

401               } else {

402                   $common = new PEAR_Common;

403                   $this->_packageXml = $common->infoFromString($contents);

404                   unset($this->_packageXml['filelist']);

405               }

406               return true;

407           } else {

408               return false;

409           }

410       }
